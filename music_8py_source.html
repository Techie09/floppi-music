48 ## Parse a string in the "music macro language"
49 #
50 # Description of the musical macro language
51 # =========================================
52 #
53 # Based on http://www.antonis.de/qbebooks/gwbasman/play.html .
54 #
55 # A-G [#,+,-]
56 # -----------
57 # A-G are notes. # or + following a note produces a sharp; - produces a
58 # flat.
59 #
60 # L(n)
61 # ----
62 # Sets the length of each note. L4 is a quarter note, L1 is a whole
63 # note, and so on. n may be from 1 to 64. Length may also follow the
64 # note to change the length for that note only. A16 is equivalent to
65 # L16A.
66 #
67 # MN
68 # --
69 # Music normal. Each note plays seven-eighths of the time determined by
70 # L (length).
71 #
72 # ML
73 # --
74 # Music legato. Each note plays the full period set by L.
75 #
76 # MS
77 # --
78 # Music staccato. Each note plays three-quarters of the time determined
79 # by L.
80 #
81 # N(n)
82 # ----
83 # Play note n. n may range from 0 to 84. In the 7 possible octaves,
84 # there are 84 notes. n set to 0 indicates a rest.
85 #
86 # O(n)
87 # ----
88 # Octave 0 sets the current octave. There are 7 octaves (0 through 6).
89 # Default is 4. Middle C is at the beginning of octave 3.
90 #
91 # P(n)
92 # ----
93 # Pause. P may range from 1-64.
94 #
95 # T(n)
96 # ----
97 # Tempo. T sets the number of L4s in a minute. n may range from 32-255.
98 # Default is 120.
99 #
100 # . (period)
101 # ----------
102 # A period after a note increases the playing time of the note by 3/2
103 # times the period determined by L (length of note) times T (tempo).
104 # Multiple periods can appear after a note, and the playing time is
105 # scaled accordingly. For example, A. will cause the note A to play one
106 # and half times the playing time determined by L (length of the note)
107 # times T (the tempo); two periods placed after A (A..) will cause the
108 # note to be played at 9/4 times its ascribed value; an A with three
109 # periods (A...) at 27/8, etc. Periods may also appear after a P
110 # (pause), and increase the pause length as described above.
111 #
112 # >n
113 # --
114 # A greater-than symbol preceding the note n plays the note in the next
115 # higher octave.
116 #
117 # <n
118 # --
119 # A less-than symbol preceding the note n plays the note in the next
120 # lower octave.
121 #
122 # |
123 # -
124 #
125 # Optionally used as a synchronization mark for multi-track music.
126 # This is a proprietary extension in the Floppi-Music project.
127 #
128 #
129 # Example
130 # =======
131 #
132 # The German children's song "Alle meine Entchen" in music macro
133 # language would be:
134 #
135 # c d e f g2 g2 a a a a g1 a a a a g1 f f f f e2 e2 d d d d c1
136 #
137 #

278 ## Parse a file in the music macro language
279 #
280 # The MML file format as used by Floppi-Music is proprietary.
281 #
282 # File format
283 # ===========
284 #
285 # Comments
286 # --------
287 #
288 # Lines starting with # are comments. At the beginning of the file,
289 # comments may be used to encode metadata. This is yet to be specified.
290 #
291 # Voices
292 # ------
293 # The voices of a song are interleaved. They are grouped per notation
294 # system, and the notation systems are seperated by empty lines.
295 #
296 #
297 # @param path the path to the MML file
298 # @return a list of lists of (frequency, duration) tuples, suitable
299 # for passing to the MusicalFloppyEnging
300 def mml_file(path):
301  vstrings = []
302  vlists = []
303  vcount = 0
304 
305  with open(path, "r") as f:
306  for l in f:
307  if l.strip().startswith("#"):
308  continue
309  elif l.strip() == "":
310  vcount = 0
311  else:
312  if len(vstrings) <= vcount:
313  vstrings.append("")
314  vstrings[vcount] += l.strip()
315  vcount += 1
316 
317  for v in vstrings:
318  vlists.append(mml(v))
319 
320  return vlists
